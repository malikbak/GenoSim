1 Overview

GenoSim addresses a critical bottleneck in clinical and population genetics research: the shortage of human NGS genotype data due to ethical, budgetary, and regulatory constraints. It provides a complete forward-time simulation pipeline that produces biologically realistic diploid SNP data across up to 10 generations.

Two simulation modes are available:

Mode Function When to use
Population simulate_population() No prior data; generate from allele frequencies
Pedigree simulate_from_pedigree() Real family VCFs + pedigree structure available

2 Installation

# From a local zip file:
install.packages("GenoSim_1.1.2.tar.gz", repos = NULL, type = "source")

3 Population Mode

3.1 Basic Simulation

The simplest use case: generate 100 founder individuals with 1,000 SNPs and run 5 generations of neutral evolution.

sim <- simulate_population(
  n_founders    = 100,
  n_snps        = 1000,
  n_generations = 5,
  chromosomes   = 1:5,
  seed          = 42,
  verbose       = FALSE
)

# Each list element is a dosage matrix [n_indiv x n_SNPs]
names(sim$genotypes)
#> [1] "gen0" "gen1" "gen2" "gen3" "gen4" "gen5"
dim(sim$genotypes[["gen0"]])
#> [1]  100 1000

3.2 Inspecting Summary Statistics

sim$summary_stats[, c("generation", "n_individuals",
                       "obs_heterozygosity", "mean_maf", "inbreeding_fis")]
#>   generation n_individuals obs_heterozygosity mean_maf inbreeding_fis
#> 1          0           100            0.36329  0.27533        0.00339
#> 2          1           150            0.36751  0.27291       -0.01640
#> 3          2           225            0.36355  0.27357       -0.00488
#> 4          3           336            0.36472  0.27425       -0.00918
#> 5          4           504            0.36670  0.27157       -0.01830
#> 6          5           756            0.36227  0.27351       -0.00326

The inbreeding_fis column tracks the Wright \(F_{IS}\) statistic:

\[F_{IS} = 1 - \frac{H_o}{H_e}\]

Under neutral panmictic evolution with no inbreeding, \(F_{IS} \approx 0\).

3.3 Consanguineous / High-Inbreeding Scenario

Set inbreeding_F = 0.125 to model a first-cousin mating system:

sim_inbred <- simulate_population(
  n_founders    = 80,
  n_snps        = 1000,
  n_generations = 5,
  inbreeding_F  = 0.125,
  n_eff         = 60,
  chromosomes   = 1:5,
  seed          = 2024,
  verbose       = FALSE
)

sim_inbred$summary_stats[, c("generation","obs_heterozygosity",
                              "exp_heterozygosity","inbreeding_fis")]

3.4 Selection + Drift

sim_sel <- simulate_population(
  n_founders    = 200,
  n_snps        = 2000,
  n_generations = 8,
  selection_s   = 0.05,    # positive selection on alt allele
  n_eff         = 150,
  chromosomes   = 1:10,
  seed          = 999,
  verbose       = FALSE
)

4 Analysis Functions

4.1 Hardy-Weinberg Equilibrium Testing

hwe <- hwe_test(sim_inbred$genotypes[[1]], alpha = 0.05)
table(hwe$sig_label)
#> 
#> HWE_deviation      HWE_pass 
#>           197           803

4.2 Linkage Disequilibrium

ld <- compute_ld(
  sim$genotypes[[1]], sim$snp_map,
  max_snps = 200, max_dist_bp = 5e6
)
cat(sprintf("Pairs: %d | Mean r²: %.4f\n", nrow(ld), mean(ld$r2, na.rm=TRUE)))
#> Pairs: 201 | Mean r²: 0.0024
plot_ld_decay(ld)

4.3 Runs of Homozygosity

roh <- detect_roh(
  sim_inbred$genotypes[[6]],
  sim_inbred$snp_map,
  min_snps      = 10,
  min_length_bp = 500000
)
nrow(roh$roh_segments)
#> [1] 10954
plot_roh_per_individual(roh)

4.4 PCA

all_genos <- do.call(rbind, sim$genotypes)
pca <- run_pca(all_genos, n_pc = 10)
cat(sprintf("PC1: %.1f%%  PC2: %.1f%%  PC3: %.1f%%\n",
            pca$variance_pct[1], pca$variance_pct[2], pca$variance_pct[3]))
#> PC1: 3.8%  PC2: 3.1%  PC3: 3.0%
gen_labels <- sub("_ind.*", "", rownames(pca$scores))
plot_pca(pca, color_by = gen_labels)

5 Pedigree Mode

5.1 Loading Example Data

GenoSim ships a complete example dataset: 25-member family FAM_KHAN across 4 generations with 4 affected individuals and consanguineous matings.

ped <- read_pedigree(example_ped_path(), verbose = FALSE)
summarise_pedigree(ped)
#> === Pedigree Summary ===
#>   Individuals : 25
#>   Families    : FAM_KHAN
#>   Per generation:
#>     Gen 0: 4 (4 founders)
#>     Gen 1: 6 (0 founders)
#>     Gen 2: 9 (0 founders)
#>     Gen 3: 6 (0 founders)
#>   Sex: 12 male | 13 female | 0 unknown
#>   Phenotype: 21 unaffected | 4 affected | 0 missing
#>   Inbreeding F: mean=0.1175 | max=0.3750
vcf <- read_vcf_cohort(example_vcf_dir(), verbose = FALSE)
cat(sprintf("Loaded: %d individuals × %d SNPs\n",
            nrow(vcf$geno_matrix), ncol(vcf$geno_matrix)))
#> Loaded: 25 individuals × 200 SNPs

5.2 Running Pedigree-Constrained Simulation

ped_sim <- simulate_from_pedigree(
  vcf_cohort        = vcf,
  pedigree          = ped,
  extra_generations = 3,
  mut_rate          = 1e-4,
  seed              = 42,
  verbose           = FALSE
)

ped_sim$summary_stats[, c("generation","n_individuals",
